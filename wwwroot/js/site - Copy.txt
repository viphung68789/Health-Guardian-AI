// ---- Auth ----
function showAuth(mode) {
    switchAuthMode(mode);
    document.getElementById('auth-form').scrollIntoView({ behavior: 'smooth' });
}

function switchAuthMode(mode) {
    const isRegister = (mode === 'register');
    // Ẩn/Hiện form tương ứng
    document.getElementById('login-form').classList.toggle('hidden', isRegister);
    document.getElementById('register-form').classList.toggle('hidden', !isRegister);
    // Cập nhật giao diện Tab (Active màu trắng)
    document.getElementById('tab-login').classList.toggle('bg-white', !isRegister);
    document.getElementById('tab-register').classList.toggle('bg-white', isRegister);
    // Cập nhật Tiêu đề
    document.getElementById('auth-title').textContent = isRegister ? 'Khởi tạo tài khoản mới' : 'Chào mừng trở lại';
}

// ---- Tab Switching ----
function switchTab(tab) {
    ['dashboard', 'profile', 'predict', 'history', 'chatbot'].forEach(t => {
        document.getElementById(`tab-${t}`).classList.toggle('active', t === tab);
        document.getElementById(`tab-${t}`).classList.toggle('hidden', t !== tab);
        document.getElementById(`nav-${t}`).classList.toggle('active', t === tab);
    });

    if (tab === 'chatbot') {
        loadChatHistory();
    }
}

// ---- Profile: BMI ----
function updateBmi() {
    const h = parseFloat(document.getElementById('profile-height').value) || 170;
    const w = parseFloat(document.getElementById('profile-weight').value) || 68;
    const bmi = (w / ((h / 100) ** 2)).toFixed(1);
    document.getElementById('live-bmi').textContent = bmi;
    let label, cls;
    if (bmi < 18.5) { label = 'Thiếu cân'; cls = 'text-sky-600 bg-sky-50 border-sky-200'; }
    else if (bmi < 25) { label = 'Bình thường'; cls = 'text-emerald-600 bg-emerald-50 border-emerald-200'; }
    else if (bmi < 30) { label = 'Thừa cân'; cls = 'text-orange-600 bg-orange-50 border-orange-200'; }
    else { label = 'Béo phì (Báo động)'; cls = 'text-red-700 bg-red-50 border-red-200'; }
    const lbl = document.getElementById('live-bmi-label');
    lbl.textContent = label;
    lbl.className = `inline-block px-3 py-1 text-xs font-extrabold border rounded-full ${cls}`;
}

}

async function loadChatHistory() {
    try {
        const response = await fetch('/Chat/GetHistory');
        if (response.ok) {
            const history = await response.json();
            const container = document.getElementById('chat-messages');
            // Giữ lại tin nhắn chào mừng đầu tiên nếu muốn, hoặc xóa hết
            container.innerHTML = ''; 
            history.forEach(msg => {
                appendChatMessage(msg.role, msg.message, msg.time);
            });
        }
    } catch (error) {
        console.error("Không thể tải lịch sử chat:", error);
    }
}

async function sendChatMessage() {
    const input = document.getElementById('chat-input');
    const msg = input.value.trim();
    if (!msg) return;
    
    input.value = '';
    appendChatMessage('user', msg);

    // Hiệu ứng đang trả lời
    const typingDiv = document.createElement('div');
    typingDiv.id = 'typing-indicator';
    typingDiv.className = 'flex gap-3 max-w-[85%] mr-auto';
    typingDiv.innerHTML = `
        <div class="w-8 h-8 rounded-lg bg-emerald-500 text-white flex items-center justify-center shrink-0 shadow-sm">
            <svg class="w-4 h-4 animate-bounce" xmlns="http://www.w3.org/2000/svg" viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2"><circle cx="12" cy="12" r="10"/></svg>
        </div>
        <div class="p-3 bg-white border border-slate-200 rounded-2xl rounded-tl-none text-xs text-slate-400 italic shadow-sm">Bác sĩ AI đang suy nghĩ...</div>
    `;
    document.getElementById('chat-messages').appendChild(typingDiv);
    document.getElementById('chat-messages').scrollTop = document.getElementById('chat-messages').scrollHeight;

    try {
        const response = await fetch('/Chat/SendMessage', {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify({ message: msg })
        });

        const typingIndicator = document.getElementById('typing-indicator');
        if (typingIndicator) typingIndicator.remove();

        if (response.ok) {
            const result = await response.json();
            appendChatMessage('model', result.response);
        } else {
            appendChatMessage('model', 'Xin lỗi, có lỗi xảy ra khi kết nối với máy chủ AI.');
        }
    } catch (error) {
        document.getElementById('typing-indicator')?.remove();
        appendChatMessage('model', 'Lỗi mạng, vui lòng kiểm tra lại kết nối.');
    }
}

function appendChatMessage(role, content, customTime = null) {
    const container = document.getElementById('chat-messages');
    const time = customTime || new Date().toLocaleTimeString('vi-VN', { hour: '2-digit', minute: '2-digit' });
    const isUser = role === 'user';
    const div = document.createElement('div');
    div.className = `flex gap-3 max-w-[85%] ${isUser ? 'ml-auto flex-row-reverse' : 'mr-auto'}`;
    const bubbleClass = isUser
        ? 'bg-indigo-600 border-indigo-700 text-white rounded-tr-none'
        : 'bg-white border-slate-200 text-slate-800 rounded-tl-none ai-message-bubble';
    div.innerHTML = `
    <div class="w-8 h-8 rounded-lg flex items-center justify-center shrink-0 shadow-sm ${isUser ? 'bg-indigo-600' : 'bg-emerald-500'} text-white">
        ${isUser
            ? `<svg class="w-4 h-4" xmlns="http://www.w3.org/2000/svg" viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2"><circle cx="12" cy="8" r="5"/><path d="M20 21a8 8 0 0 0-16 0"/></svg>`
            : `<svg class="w-4 h-4" xmlns="http://www.w3.org/2000/svg" viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2"><path d="M19 14c1.49-1.46 3-3.21 3-5.5A5.5 5.5 0 0 0 16.5 3c-1.76 0-3 .5-4.5 2-1.5-1.5-2.74-2-4.5-2A5.5 5.5 0 0 0 2 8.5c0 2.3 1.5 4.05 3 5.5l7 7Z"/></svg>`
        }
    </div>
    <div class="space-y-1">
        <div class="p-3.5 rounded-2xl border text-l leading-relaxed ${isUser ? 'bg-indigo-600 border-indigo-700 text-white rounded-tr-none' : 'bg-white border-slate-200 text-slate-800 rounded-tl-none'}">${content}</div>
        <span class="block text-[0.625rem] text-slate-400 font-mono ${isUser ? 'text-right' : 'text-left'}">${time}</span>
    </div>`;

    const bubble = div.querySelector('.rounded-2xl')
    if (isUser) {
        bubble.textContent = content; // User input: plain text, tránh XSS
    } else {
        bubble.innerHTML = content;   // AI response: đã là HTML từ server
    }
    container.appendChild(div);
    container.scrollTop = container.scrollHeight;
}

async function predictStroke() {
    //Tran lết Tiếng ziệt
    const featureNames = {
        Age: "Tuổi",
        Hypertension: "Tăng huyết áp",
        Heart_Disease: "Bệnh tim",
        BMI: "BMI",
        Avg_Glucose: "Đường huyết trung bình",
        Diabetes: "Tiểu đường",
        Gender: "Giới tính",
        SES: "Tình trạng kinh tế - xã hội",
        Smoking_Status: "Tình trạng hút thuốc"
    };
    const request = {
        Age: parseFloat(document.getElementById('profile-age').value),
        Hypertension: hypertension ? 1 : 0,
        Heart_Disease: heartDisease ? 1 : 0,
        BMI: parseFloat(
            document.getElementById('live-bmi').textContent
        ),
        Avg_Glucose: parseFloat(
            document.getElementById('num-glucose').value
        ),
        Diabetes: diabetes ? 1 : 0,
        Gender: document.getElementById('profile-gender').value,
        SES: document.getElementById('profile-ses').value,

        Smoking_Status:
            document.getElementById('profile-smoking').value
    };
    const response = await fetch('/Dashboard/Predict', {
        method: 'POST',
        headers: {
            'Content-Type': 'application/json'
        },
        body: JSON.stringify(request)
    });
    const result = await response.json();
    let html = `
    <h3>Kết quả AI</h3>
    <p>Dự đoán: ${result.prediction}</p>
    <p>Xác suất: ${(result.probability * 100).toFixed(2)}%</p>`;
    if (result.top_factors) {
        html += "<h4>Yếu tố nguy cơ</h4><ul>";
        result.top_factors.forEach(x => { const displayName = featureNames[x.feature] || x.feature; html += `<li>${displayName}</li>`; });
        html += "</ul>";
    }
    document.getElementById("predict-result").innerHTML = html;
}
